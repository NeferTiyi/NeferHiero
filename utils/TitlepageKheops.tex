\newlength{\drop}% for my convenience

\makeatletter

  \def\maketitle{%
    \drop = 0.1\textheight
    \vfill%
    \hbox{%
      \hspace*{0.2\textwidth}%
      \rule{1pt}{\textheight}
      \hspace*{0.05\textwidth}%
      \parbox[b]{0.75\textwidth}{
        \vbox to 0.9\textheight{%
          \vspace{\drop}
          {\noindent\HUGE\bfseries \@title\par}
          \vspace{2\baselineskip}
          {\Huge\itshape \@subtitle\par}
          \vspace{4\baselineskip}
          {\Large\@orga\\[0.5\baselineskip]\@lecturer\par}

          \vspace*{\fill}

          {\noindent \@author\par}
          \bigskip
          {\small\itshape\today, \printtime\par}
        }% end of vbox
      }% end of parbox
    }% end of hbox
    \vfill%

    \pagebreak

    \thispagestyle{empty}
    \vspace*{\fill}
    \input{\DirUtils/licence}
    Licence
    \pagebreak
  }

\makeatother
