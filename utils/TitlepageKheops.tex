      \hspace*{0.05\textwidth}%

        \vbox{%
          \vspace{\drop}
          {\noindent\HUGE\bfseries \@title}\\[2\baselineskip]

                  \@lecturer}\par

          {\noindent \@author}\\[\baselineskip]
          {\small\itshape\today, \printtime}
        }% end of vbox
      }% end of parbox
    }% end of hbox
    \vfill%

    \pagebreak

    \thispagestyle{empty}
    \vspace*{\fill}
    \input{../../../utils/licence}
    \pagebreak
  }
